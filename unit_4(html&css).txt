### 2. Registration Form with Validation:

<!DOCTYPE html>
<html>
<body>

<form onsubmit="return validate()">
  Name: <input type="text" id="name"><br>
  Email: <input type="text" id="email"><br>
  <input type="submit">
</form>

<script>
function validate() {
  let name = document.getElementById("name").value;
  let email = document.getElementById("email").value;

  if (name == "" || email == "") {
    alert("Fill all fields");
    return false;
  }
  return true;
}
</script>

</body>
</html>

### 4. To-Do List (Add/Delete):

<!DOCTYPE html>
<html>
<body>

<input id="task">
<button onclick="addTask()">Add</button>
<ul id="list"></ul>

<script>
function addTask() {
  let li = document.createElement("li");
  li.innerText = document.getElementById("task").value;

  li.onclick = function() { this.remove(); }

  document.getElementById("list").appendChild(li);
}
</script>

</body>
</html>

### 6. Email & Password Validation:

<!DOCTYPE html>
<html>
<body>

<input id="email" placeholder="Email"><br>
<input id="pass" placeholder="Password"><br>
<button onclick="check()">Submit</button>

<script>
function check() {
  let email = document.getElementById("email").value;
  let pass = document.getElementById("pass").value;

  if (!email.includes("@"))
    alert("Invalid Email");
  else if (pass.length < 6)
    alert("Weak Password");
  else
    alert("Valid");
}
</script>

</body>
</html>
